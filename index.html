拼接字符串

给定A,B两个字符串，串中的字符全都是小写英文字母（即a -> z）。判断能否只使用A串中的字符拼出B串，A串中的每个字符只准使用一次。例如：

A = “aaab” B = “aaa” true

A = “aabb” B= “aaa” false

A = “aaab” B = “abaa” true

A = “aaab” B = “aaac” false

函数原型：

最佳答案

// 计算每个字母出现的次数，正向加法，逆向加法。小于0证明B比A中同样的字母少
BOOL canConstruct(const char A, const char B);
BOOL canConstruct(const char A, const char B){

int a[26] = {};
for (int i = 0; i < strlen(A); i ++) {
    a[A[i] - 'a'] += 1;
}
for (int j = 0; j < strlen(B); j ++) {
    a[B[j] - 'a'] -= 1;
    if (a[B[j] - 'a'] < 0) {
        return false;
    }
}
return true;

}
int main(int argc, const char * argv[]) {
@autoreleasepool {
char a[] = “aaab”;
char b[] = “aaca”;
printf(“%d”,canConstruct(a, b));
}
return 0;
}

每周烧脑-002

发表于 2016-09-26   |   分类于 Algorithm   |     |   阅读次数

题目

螺旋二维数组

输入：正整数N

输出：N x N的二维数组，要求，数组元素从1到NxN螺旋排列，如N=5时，输出如下结果：

1 2 3 4 5

16 17 18 19 6

15 24 25 20 7

14 23 22 21 8

13 12 11 10 9

最佳答案

#import

#define M 100

#define N 100
/
填充方法，归纳每一行的数据规律
k为当前放下的数字，一边走一边放
按照摆放数据的过程，逐层分别处理每圈的上下左右方的数据 /
int main(int argc, const char * argv[]) {

int i, j, a[100][100]={0}, n, k = 1;
printf("please input a number n:");
while(scanf("%d", &n)!=EOF){
    k = 1;
    for (i = 1; i <= n / 2; ++ i){
        for (j = i; j <= n - i; ++ j) /* 左侧 */
            a[i][j] = k ++;
        for (j = i; j <= n - i; ++ j) /* 下方 */
            a[j][n + 1 - i] = k ++;
        for (j = n - i + 1; j >= i + 1; j = j - 1) /* 右侧 */
            a[n + 1 - i][j] = k ++;
        for (j = n - i + 1; j >= i + 1; j = j - 1) /* 上方 */
            a[j][i] = k ++;
    }
    if(n != 1){
        for (i = 1; i <= n; ++ i){
            for (j = 1; j <= n; ++ j)
                printf("%5d", a[i][j]); printf("\n");
        }
    }
    else
        printf("%5d\n",n);
    printf("please input a number n:");
}

return 0;

}
